```py
def get_movie(id):
    try:
        pipeline = [
            {
                "$match": {
                    "_id": ObjectId(id)
                }
            },
            {
                "$lookup": {
                    "from": "comments",
                    "localField": "_id",
                    "foreignField": "movie_id",
                    "as": "comments"
                }
            }
        ]
        movie = db.movies.aggregate(pipeline).next()
        movie["comments"] = sorted(
            movie.get("comments", []),
            key=lambda c: c.get("date"),
            reverse=True
        )
        return movie
    # this will handle "InvalidId" the same way as "StopIteration"
    except (InvalidId, StopIteration) as _:
        return None
```
